export function pollMidi(): MidiMessage[] {
	if (!midiIn) return [];
	const messages: MidiMessage[] = [];
	const recorded = midiHeaderView.getUint32(12, true); // dwBytesRecorded
	if (recorded === 0) return messages;
	// Each short message is 4 bytes (3 used)
	for (let offset = 0; offset + 3 < recorded; offset += 4) {
		const status = midiBuffer[offset];
		const data1 = midiBuffer[offset + 1];
		const data2 = midiBuffer[offset + 2];
		messages.push({ status, data1, data2 });
	}
	// clear MHDR_DONE
	midiHeaderView.setUint32(20, 0, true);
	// Reset buffer and re‑add it
	midiHeaderView.setUint32(12, 0, true);
	winmm.symbols.midiInAddBuffer(midiIn, Deno.UnsafePointer.of(midiHeaderBuffer), 64);
	return messages;
}

export function initMidiPoll(deviceId: number): boolean {
  if (winmm.symbols.midiInGetNumDevs() === 0) return false;

  const outHandleBuffer = new BigUint64Array(1);
  
  // Open with CALLBACK_NULL so no background callback thread is generated
  const openResult = winmm.symbols.midiInOpen(
    outHandleBuffer,
    deviceId,
    null,
    null,
    CALLBACK_NULL
  );
  if (openResult !== 0) return false;

  // Extract pointer reliably using Deno's updated UnsafePointer mapping
  hMidiIn = Deno.UnsafePointer.create(outHandleBuffer[0]);

  // Set up MIDIHDR layout pointer destinations manually 
  // Offset 0: Pointer to our raw data buffer
  const dataPtr = Deno.UnsafePointer.of(midiDataBuffer);
  headerView32.setBigUint64(0, BigInt(Deno.UnsafePointer.value(dataPtr)), true);
  // Offset 8: Buffer capacity length (4 bytes)
  headerView32.setUint32(8, 4, true);

  // Prepare and inject the tracking buffer state into Windows mm system
  winmm.symbols.midiInPrepareHeader(hMidiIn, midiHeaderBuffer, midiHeaderBuffer.byteLength);
  winmm.symbols.midiInAddBuffer(hMidiIn, midiHeaderBuffer, midiHeaderBuffer.byteLength);
  
  return winmm.symbols.midiInStart(hMidiIn) === 0;
}

/**
 * Non-blocking check for pending MIDI updates.
 * Call this inside your main animation or processing loop.
 */
export function pollMidi(): MidiMessage | null {
  if (!hMidiIn) return null;

  // Read the 32-bit flags register located at offset 16 inside MIDIHDR struct
  const flags = headerView32.getUint32(16, true);

  // Check if Windows marked the buffer block as complete (MHDR_DONE)
  if ((flags & MHDR_DONE) === MHDR_DONE) {
    // Extract standard MIDI values directly from the assigned byte array
    const status = midiDataBuffer[0];
    const data1 = midiDataBuffer[1];
    const data2 = midiDataBuffer[2];

    // Reset the internal Windows state flags inside our structure mapping
    headerView32.setUint32(16, flags & ~MHDR_DONE, true);

    // Re-queue the tracking block back into the buffer stream loop
    winmm.symbols.midiInAddBuffer(hMidiIn, midiHeaderBuffer, midiHeaderBuffer.byteLength);

    return { status, data1, data2 };
  }

  return null; // Return null if no new MIDI action occurred during this tick
}
